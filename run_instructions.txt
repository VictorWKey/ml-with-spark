# Instrucciones para ejecutar los componentes del proyecto (ejecutar desde la raíz del workspace)

# 1) Ejecutar spark-shell de forma interactiva (recomendado para debug/inspección)
# Ajusta la memoria según tu máquina
spark-shell --driver-memory 14g --executor-memory 14g \
	--conf spark.memory.fraction=0.8 \
	--conf spark.memory.storageFraction=0.2 \
	--conf spark.sql.shuffle.partitions=100

spark-shell --driver-memory 4g --executor-memory 4g --conf spark.memory.fraction=0.8 --conf spark.memory.storageFraction=0.2 --conf spark.sql.shuffle.partitions=100

# Dentro de spark-shell, puedes cargar y ejecutar cada script así:
# - Validación de datos
:load ml-prediction/DataValidation.scala
DataValidation.main(Array())

# - Modelo REAL (sin target encoding)
:load ml-prediction/IMDBPredictionModelClean.scala
IMDBPredictionModelClean.main(Array())

# - Modelo Simplified (con target encoding y pipeline optimizado)
:load ml-prediction/IMDBPredictionModelSimplified.scala
IMDBPredictionModelSimplified.main(Array())
